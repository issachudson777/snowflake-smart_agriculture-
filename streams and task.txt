USE WAREHOUSE AGRICULTURE_WH;
USE DATABASE AGRICULTURE_DB;
USE SCHEMA RAW;

-- =========================================
-- STREAMS
-- =========================================

CREATE OR REPLACE STREAM FARMER_STREAM
ON TABLE FARMER;

CREATE OR REPLACE STREAM CROP_STREAM
ON TABLE CROP;

CREATE OR REPLACE STREAM FARMPRODUCTION_STREAM
ON TABLE FARMPRODUCTION;

CREATE OR REPLACE STREAM FERTILIZER_STREAM
ON TABLE FERTILIZERUSAGE;

CREATE OR REPLACE STREAM YIELDALERT_STREAM
ON TABLE YIELDALERT;

CREATE OR REPLACE STREAM RAW_FERTILIZER_JSON_STREAM
ON TABLE RAW_FERTILIZER_JSON;

CREATE OR REPLACE STREAM RAW_YIELDALERT_JSON_STREAM
ON TABLE RAW_YIELDALERT_JSON;

-- =========================================
-- JSON FLATTEN TASKS (auto-flatten new JSON into relational tables)
-- =========================================

CREATE OR REPLACE TASK TASK_FLATTEN_FERTILIZER_JSON
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO AGRICULTURE_DB.RAW.FertilizerUsage
SELECT
    VALUE:usageId::INT,
    VALUE:cropId::INT,
    VALUE:fertilizerName::VARCHAR(100),
    VALUE:quantityUsed::NUMBER(10,2),
    VALUE:usageDate::DATE
FROM AGRICULTURE_DB.RAW.RAW_FERTILIZER_JSON_STREAM,
LATERAL FLATTEN(INPUT => RAW_DATA);

CREATE OR REPLACE TASK TASK_FLATTEN_YIELDALERT_JSON
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO AGRICULTURE_DB.RAW.YieldAlert
SELECT
    VALUE:alertId::INT,
    VALUE:cropId::INT,
    VALUE:alertReason::VARCHAR(255),
    VALUE:yieldScore::NUMBER(5,2),
    VALUE:alertStatus::VARCHAR(50)
FROM AGRICULTURE_DB.RAW.RAW_YIELDALERT_JSON_STREAM,
LATERAL FLATTEN(INPUT => RAW_DATA);

-- =========================================
-- RAW TO CORE TASKS
-- =========================================

USE SCHEMA CORE;

CREATE OR REPLACE TASK TASK_FARMER_RAW_TO_CORE
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO Farmer_RAW_TO_CORE
SELECT farmerId, farmerName, village, district, contactNumber, farmerStatus
FROM AGRICULTURE_DB.RAW.FARMER_STREAM
WHERE METADATA$ACTION = 'INSERT'
AND farmerStatus <> 'INACTIVE';

CREATE OR REPLACE TASK TASK_CROP_RAW_TO_CORE
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO Crop_RAW_TO_CORE
SELECT cropId, cropName, cropCategory, season
FROM AGRICULTURE_DB.RAW.CROP_STREAM
WHERE METADATA$ACTION = 'INSERT';

CREATE OR REPLACE TASK TASK_FARMPRODUCTION_RAW_TO_CORE
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO FarmProduction_RAW_TO_CORE
SELECT productionId, farmerId, cropId, productionQuantity, harvestDate
FROM AGRICULTURE_DB.RAW.FARMPRODUCTION_STREAM
WHERE METADATA$ACTION = 'INSERT'
AND productionQuantity > 0;

CREATE OR REPLACE TASK TASK_FERTILIZER_RAW_TO_CORE
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO FertilizerUsage_RAW_TO_CORE
SELECT usageId, cropId, fertilizerName, quantityUsed, usageDate
FROM AGRICULTURE_DB.RAW.FERTILIZER_STREAM
WHERE METADATA$ACTION = 'INSERT'
AND quantityUsed >= 0;

CREATE OR REPLACE TASK TASK_YIELDALERT_RAW_TO_CORE
WAREHOUSE = AGRICULTURE_WH
SCHEDULE = '5 MINUTE'
AS
INSERT INTO YieldAlert_RAW_TO_CORE
SELECT alertId, cropId, alertReason, yieldScore, alertStatus
FROM AGRICULTURE_DB.RAW.YIELDALERT_STREAM
WHERE METADATA$ACTION = 'INSERT';

-- =========================================
-- RESUME TASKS
-- =========================================

ALTER TASK AGRICULTURE_DB.RAW.TASK_FLATTEN_FERTILIZER_JSON RESUME;
ALTER TASK AGRICULTURE_DB.RAW.TASK_FLATTEN_YIELDALERT_JSON RESUME;
ALTER TASK TASK_FARMER_RAW_TO_CORE RESUME;
ALTER TASK TASK_CROP_RAW_TO_CORE RESUME;
ALTER TASK TASK_FARMPRODUCTION_RAW_TO_CORE RESUME;
ALTER TASK TASK_FERTILIZER_RAW_TO_CORE RESUME;
ALTER TASK TASK_YIELDALERT_RAW_TO_CORE RESUME;
