USE WAREHOUSE AGRICULTURE_WH;
USE DATABASE AGRICULTURE_DB;
USE SCHEMA CORE;

CREATE OR REPLACE TABLE Farmer_RAW_TO_CORE (
    farmerId INT PRIMARY KEY,
    farmerName VARCHAR(100),
    village VARCHAR(100),
    district VARCHAR(100),
    contactNumber VARCHAR(20),
    farmerStatus VARCHAR(20)
);

CREATE OR REPLACE TABLE Crop_RAW_TO_CORE (
    cropId INT PRIMARY KEY,
    cropName VARCHAR(100),
    cropCategory VARCHAR(100),
    season VARCHAR(50)
);

CREATE OR REPLACE TABLE FarmProduction_RAW_TO_CORE (
    productionId INT PRIMARY KEY,
    farmerId INT,
    cropId INT,
    productionQuantity NUMBER(10,2)
        CHECK(productionQuantity > 0),
    harvestDate DATE,

    FOREIGN KEY (farmerId)
        REFERENCES Farmer_RAW_TO_CORE(farmerId),

    FOREIGN KEY (cropId)
        REFERENCES Crop_RAW_TO_CORE(cropId)
);

CREATE OR REPLACE TABLE FertilizerUsage_RAW_TO_CORE (
    usageId INT PRIMARY KEY,
    cropId INT,
    fertilizerName VARCHAR(100),
    quantityUsed NUMBER(10,2)
        CHECK(quantityUsed >= 0),
    usageDate DATE,

    FOREIGN KEY (cropId)
        REFERENCES Crop_RAW_TO_CORE(cropId)
);

CREATE OR REPLACE TABLE YieldAlert_RAW_TO_CORE (
    alertId INT PRIMARY KEY,
    cropId INT,
    alertReason VARCHAR(255),
    yieldScore NUMBER(5,2),
    alertStatus VARCHAR(50),

    FOREIGN KEY (cropId)
        REFERENCES Crop_RAW_TO_CORE(cropId)
);


INSERT INTO Farmer_RAW_TO_CORE
SELECT DISTINCT *
FROM RAW.Farmer
WHERE farmerStatus <> 'INACTIVE';

INSERT INTO Crop_RAW_TO_CORE
SELECT DISTINCT *
FROM RAW.Crop;

INSERT INTO FarmProduction_RAW_TO_CORE
SELECT DISTINCT *
FROM RAW.FarmProduction
WHERE productionQuantity > 0;

INSERT INTO FertilizerUsage_RAW_TO_CORE
SELECT DISTINCT *
FROM RAW.FertilizerUsage
WHERE quantityUsed >= 0;

INSERT INTO YieldAlert_RAW_TO_CORE
SELECT DISTINCT *
FROM RAW.YieldAlert;
