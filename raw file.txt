USE WAREHOUSE AGRICULTURE_WH;
USE DATABASE AGRICULTURE_DB;
USE SCHEMA RAW;


CREATE OR REPLACE TABLE Farmer (
    farmerId INT,
    farmerName VARCHAR(100),
    village VARCHAR(100),
    district VARCHAR(100),
    contactNumber VARCHAR(20),
    farmerStatus VARCHAR(20)
);

CREATE OR REPLACE TABLE Crop (
    cropId INT,
    cropName VARCHAR(100),
    cropCategory VARCHAR(100),
    season VARCHAR(50)
);

CREATE OR REPLACE TABLE FarmProduction (
    productionId INT,
    farmerId INT,
    cropId INT,
    productionQuantity NUMBER(10,2),
    harvestDate DATE
);

CREATE OR REPLACE TABLE FertilizerUsage (
    usageId INT,
    cropId INT,
    fertilizerName VARCHAR(100),
    quantityUsed NUMBER(10,2),
    usageDate DATE
);


CREATE OR REPLACE TABLE YieldAlert (
    alertId INT,
    cropId INT,
    alertReason VARCHAR(255),
    yieldScore NUMBER(5,2),
    alertStatus VARCHAR(50)
);

-- FILE FORMATS
CREATE OR REPLACE FILE FORMAT CSV_FORMAT
TYPE=CSV
SKIP_HEADER=1
FIELD_OPTIONALLY_ENCLOSED_BY='"'
EMPTY_FIELD_AS_NULL=TRUE
TRIM_SPACE=TRUE;

CREATE OR REPLACE FILE FORMAT JSON_FORMAT
TYPE=JSON;



-- STAGES
CREATE OR REPLACE STAGE CSV_STAGE
  URL = 's3://final-mock-agri-bucket/CSV/'
  CREDENTIALS = (
    AWS_KEY_ID = 'AWS KEY ENTER HERE'
    AWS_SECRET_KEY = 'AWS SECRET KEY ENTER HERE'
  )
  FILE_FORMAT = CSV_FORMAT;

CREATE OR REPLACE STAGE JSON_STAGE
  URL = 's3://final-mock-agri-bucket/JSON/'
  CREDENTIALS = (
    AWS_KEY_ID = 'AWS KEY ENTER HERE'
    AWS_SECRET_KEY = 'AWS SECRET KEY ENTER HERE'
  )
  FILE_FORMAT = JSON_FORMAT;


-- RAW JSON TABLES
CREATE OR REPLACE TABLE RAW_FERTILIZER_JSON(
RAW_DATA VARIANT
);

CREATE OR REPLACE TABLE RAW_YIELDALERT_JSON(
RAW_DATA VARIANT
);


-- PIPES
CREATE OR REPLACE PIPE FARMER_PIPE
AUTO_INGEST=TRUE
AS
COPY INTO Farmer
FROM @CSV_STAGE
FILE_FORMAT=CSV_FORMAT
PATTERN='.*FARMER.*[.]csv';

CREATE OR REPLACE PIPE CROP_PIPE
AUTO_INGEST=TRUE
AS
COPY INTO Crop
FROM @CSV_STAGE
FILE_FORMAT=CSV_FORMAT
PATTERN='.*CROP.*[.]csv';

CREATE OR REPLACE PIPE FARMPRODUCTION_PIPE
AUTO_INGEST=TRUE
AS
COPY INTO FarmProduction
FROM @CSV_STAGE
FILE_FORMAT=CSV_FORMAT
PATTERN='.*FARMPRODUCTION.*[.]csv';

CREATE OR REPLACE PIPE FERTILIZER_PIPE
AUTO_INGEST=TRUE
AS
COPY INTO RAW_FERTILIZER_JSON
FROM @JSON_STAGE
FILE_FORMAT=JSON_FORMAT
PATTERN='.*FERTILIZER.*[.]json';

CREATE OR REPLACE PIPE YIELDALERT_PIPE
AUTO_INGEST=TRUE
AS
COPY INTO RAW_YIELDALERT_JSON
FROM @JSON_STAGE
FILE_FORMAT=JSON_FORMAT
PATTERN='.*YIELDALERT.*[.]json';


-- REFRESH PIPES
ALTER PIPE FARMER_PIPE REFRESH;
ALTER PIPE CROP_PIPE REFRESH;
ALTER PIPE FARMPRODUCTION_PIPE REFRESH;
ALTER PIPE FERTILIZER_PIPE REFRESH;
ALTER PIPE YIELDALERT_PIPE REFRESH;

-- =========================================

INSERT INTO FertilizerUsage
SELECT
    VALUE:usageId::INT,
    VALUE:cropId::INT,
    VALUE:fertilizerName::VARCHAR(100),
    VALUE:quantityUsed::NUMBER(10,2),
    VALUE:usageDate::DATE
FROM RAW_FERTILIZER_JSON,
LATERAL FLATTEN(INPUT => RAW_DATA);

INSERT INTO YieldAlert
SELECT
    VALUE:alertId::INT,
    VALUE:cropId::INT,
    VALUE:alertReason::VARCHAR(255),
    VALUE:yieldScore::NUMBER(5,2),
    VALUE:alertStatus::VARCHAR(50)
FROM RAW_YIELDALERT_JSON,
LATERAL FLATTEN(INPUT => RAW_DATA);
